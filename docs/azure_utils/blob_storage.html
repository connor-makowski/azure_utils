 85        self, remote_filepath: str, local_filepath: str, overwrite: bool = False
 86    ) -> None:
 87        """
 88        Function:
 89
 90        - Uploads a file from local to remote
 91
 92        Parameters:
 93
 94        - `remote_filepath` (str): The path to the remote file
 95        - `local_filepath` (str): The path to the local file
 96        - `overwrite` (bool): Whether to overwrite the remote file if it already exists
 97            - Optional: Defaults to `False`
 98
 99        Returns:
100
101        - `None`
102
103        Example:
104
105        ```python
106        from azure_utils.blob_storage import AZContainer

111        )
112        mycontainer.upload_file(
113            remote_filepath='/path/to/file.txt',
114            local_filepath='/path/to/file.txt',
115            overwrite=True
116        )
117        # This will upload the local file `/path/to/file.txt` to the remote file `/path/to/file.txt`
118        ```

124        with open(local_filepath, "rb") as data:
125            blob = MetaBlob(
126                blob_client=self.client.upload_blob(
127                    name=remote_filepath, data=data, overwrite=overwrite
128                ),
129                filepath=local_filepath,
130            )
131        blob.update_meta()

221    def sync_to_local(
222        self,
223        remote_folderpath: str,
224        local_folderpath: str,
225        overwrite: bool = False,
226        smart_sync: bool = False,
227        removal: bool = False,
228    ) -> None:
229        """
230        Function:
231
232        - Sync all files from a remote folder to a local folder
233
234        Parameters:
235
236        - `remote_folderpath` (str): The path to the remote folder
237        - `local_folderpath` (str): The path to the local folder
238        - `overwrite` (bool): Whether to overwrite the local file if it already exists
239            - Optional: Defaults to `False`
240        - `smart_sync` (bool): Whether to skip downloading if the remote etag and md5 hash match the local meta file
241            - Optional: Defaults to `False`
242        - `removal` (bool): Whether to remove local files that do not exist in the remote folder
243            - Optional: Defaults to `False`
244
245        Returns:
246
247        - `None`
248
249        Example:
250
251        ```python
252        from azure_utils.blob_storage import AZContainer
253        mycontainer = AZContainer(
254            account_url="my_account_url",
255            account_key="my_account_key",
256            container_name="my_container"
257        )
258        mycontainer.sync_to_local(
259            remote_folderpath='/path/to/folder/',
260            local_folderpath='/path/to/folder/',
261            overwrite=True,
262            smart_sync=True,
263            removal=True
264        )
265        # This will sync the remote folder `/path/to/folder/` to the local folder `/path/to/folder/`
266        # It will also remove any local files that do not exist in the remote folder
267        ```
268        """
269        validate_path(path=local_folderpath, is_remote=False, is_folder=True)
270        validate_path(path=remote_folderpath, is_remote=True, is_folder=True)
271        blobs = [
272            i
273            for i in self.client.list_blobs(name_starts_with=remote_folderpath)
274        ]
275        for blobProperty in blobs:
276            blob = MetaBlob(
277                blob_client=None,
278                filepath=local_folderpath
279                + blobProperty.name.replace(remote_folderpath, "", 1),
280                smart_sync=smart_sync,
281                remote_etag=blobProperty.etag,
282                overwrite=overwrite,
283            )
284            if blob.block_download:
285                continue
286            blob.blob_client = self.client.get_blob_client(
287                blob=blobProperty.name
288            )
289            blob.download()
290        if removal:
291            posix_folderpath = str(pathlib.PurePosixPath(local_folderpath))
292            if posix_folderpath == "/":
293                raise Exception("Cannot do removal in the root folder")
294            storage_filepaths = set(
295                [i.name.replace(remote_folderpath, "") for i in blobs]
296            )
297            for local_filepath in [
298                str(i) for i in pathlib.Path(local_folderpath).glob("**/*")
299            ]:
300                posix_filepath = str(pathlib.PurePosixPath(local_filepath))
301                if (
302                    posix_filepath.replace(posix_folderpath, "").replace(
303                        ".meta.", ""
304                    )
305                    not in storage_filepaths
306                ):
307                    os.remove(local_filepath)

309    def sync_to_remote(
310        self,
311        remote_folderpath: str,
312        local_folderpath: str,
313        overwrite: bool = False,
314        omissions: list = [".meta."],
315    ) -> None:
316        """
317        Function:
318
319        - Sync all files from a local folder to a remote folder
320
321        Parameters:
322
323        - `remote_folderpath` (str): The path to the remote folder
324        - `local_folderpath` (str): The path to the local folder
325        - `overwrite` (bool): Whether to overwrite the remote file if it already exists
326            - Optional: Defaults to `False`
327        - `omissions` (list): A list of strings to omit from the upload
328            - Optional: Defaults to `['.meta.']`
329
330        Returns:
331
332        - `None`
333
334        Example:
335        ```python
336        from azure_utils.blob_storage import AZContainer
337        mycontainer = AZContainer(
338            account_url="my_account_url",
339            account_key="my_account_key",
340            container_name="my_container"
341        )
342        mycontainer.sync_to_remote(
343            remote_folderpath='/path/to/folder/',
344            local_folderpath='/path/to/folder/',
345        )
346        # This will sync the local folder `/path/to/folder/` to the remote folder `/path/to/folder/`
347        ```
348        """
349        validate_path(path=local_folderpath, is_remote=False, is_folder=True)
350        validate_path(path=remote_folderpath, is_remote=True, is_folder=True)
351        local_filepaths = [
352            str(i)
353            for i in pathlib.Path(local_folderpath).glob("**/*")
354            if i.is_file()
355        ]
356        local_filepaths = [
357            i for i in local_filepaths if not any([j in i for j in omissions])
358        ]
359        for local_filepath in local_filepaths:
360            self.upload_file(
361                remote_filepath=remote_folderpath
362                + local_filepath.replace(local_folderpath, ""),
363                local_filepath=local_filepath,
364                overwrite=overwrite,
365            )
